public int trap(int[] height) {

	if(height == null || height.length ==0) return 0;
	
	int p1=0,p2=height.length-1;
	
	int lMax = height[p1] , rMax = height[p2];
	
	int res = 0;
	
	while(p1<p2)
	
	{
		
		if(lMax<rMax)
		
		{
		
			p1++;
			
			lMax = Math.max(lMax,height[p1]);
			
			res += lMax-height[p1];
		
		} else {
		
			p2--;
			
			rMax = Math.max(rMax,height[p2]);
			
			res += rMax-height[p2];
		
		}
	
	}
	
	return res;

}
